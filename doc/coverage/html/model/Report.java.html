package model;


import java.util.HashMap;
import java.util.Map;

public class Report {

    Map<ReportCodeKey, String> valueMap;

    public Report() {
        valueMap = new HashMap<>();
        for (ReportCodeKey key : ReportCodeKey.values()) {
            valueMap.put(key, null);
        }
    }

    public Map<String, String> valuesToMap() {
        Map<String, String> map = new HashMap<>();
        for (Map.Entry<ReportCodeKey, String> entry : valueMap.entrySet()) {
            map.put(entry.getKey().getStringValue(), entry.getValue());
        }
        return map;
    }

    public String getValueOf(ReportCodeKey reportCodeKey) {
        return valueMap.get(reportCodeKey);
    }

    public void setValueOf(ReportCodeKey reportCodeKey, String value) {
        if (valueMap.containsKey(reportCodeKey)) {
            valueMap.put(reportCodeKey, value);
        } else {
            throw new IllegalArgumentException("No such value in report!");
        }
    }

    public boolean isValid() {
        Map<String, String> mapper = valuesToMap();
        for (Map.Entry <String,String> entry : valuesToMap().entrySet()) {
            if (entry.getValue() == null) {
                System.out.println("Missing key: " + entry.getKey());
                return false;
            }
        }
        return true;
    }

    /**
     * @return Condition required for returning true in isValid
     */
    public String getValidConditions() {
        String validatorCondition = "Following values must exist in mapper and none of existing pairs can be null:\n";
        for (ReportCodeKey key : valueMap.keySet()) {
            validatorCondition += key.getStringValue() + "\n";
        }
        return validatorCondition;
    }

}
