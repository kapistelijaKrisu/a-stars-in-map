package model.structure;

public enum CharForTypeOfProcessing {
    IS_START_AND_TARGET('O'),
    START_LOCATION('S'),
    TARGET_AND_NOT_FOUND('N'),
    TARGET_AND_FOUND('F'),
    NOT_PROCESSED('.'), PROCESSED_NOT_IN_PATH('v'),
    PROCESSED_IN_PATH('X'),
    WALL('#');

    private final char nodeType;

    CharForTypeOfProcessing(char nodeType) {
        this.nodeType = nodeType;
    }

    public char getNodeTypet() {
        return this.nodeType;
    }
}
